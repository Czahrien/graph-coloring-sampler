\begin{itemize}

\item $\displaystyle\rho_i = \frac{\mbox{number of colorings in } G_{i+1} }{\mbox{number of colorings in } G_{i} } = \frac{|G_{i+1}|}{|G_{i}|}$

\item $\displaystyle\prod_{i=1}^{n-1}{\rho_i} = \frac{|C(G_n)|}{q^n}$
\end{itemize}
